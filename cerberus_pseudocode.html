// ── CONSTANTS ──────────────────────────────────────────────────────────
DEFINE HONEY_FILES = { passwords.txt, confidential.docx, financial_budget_2024.xlsx, ... } // 10 decoys
DEFINE RISK_SCORES = { OPENED:10, MODIFIED:40, MOVED:60, COPIED:70, DELETED:90, SMB_GET:80, SMB_PUSH:90 }
DEFINE LOG_FILE = "logs/alerts.log"
// ── HELPER FUNCTIONS ────────────────────────────────────────────────────
FUNCTION calculate_severity(score):
  IF score < 20 → RETURN "LOW"  |  IF score < 40 → RETURN "HIGH"  |  ELSE → RETURN "CRITICAL"
FUNCTION send_critical_alert(severity, action, file_name, location):
  SEND email via Gmail SMTP (port 587, STARTTLS, App Password)
  SEND Telegram message via HTTP POST /sendMessage (HTML formatted, attacker IP + filename)
FUNCTION log_event(action, location, file_name):
  IF active SMB IPs exist AND action IN [OPENED, MODIFIED, COPIED]:
    Upgrade action → SMB_PUSH / SMB_GET // remote session escalation
  risk_score = RISK_SCORES[action]  |  severity = calculate_severity(risk_score)
  APPEND timestamped record to LOG_FILE (user, action, file, severity, score) + fsync
  IF severity == "CRITICAL" → SPAWN thread → send_critical_alert()
// ── DETECTION THREADS ───────────────────────────────────────────────────
FUNCTION monitor_security_logs(): // Thread 1 — Windows Event 5145
  REQUIRE Administrator privileges | Enable audit policy: Detailed File Share
  LOOP every 2 s:
    Query Windows Security Log for Event ID 5145 (wevtutil XML query)
    FOR EACH new event: Extract IpAddress + RelativeTargetName from XML
      IF file matches HONEY_FILES AND debounce(3 s) passes:
        CALL log_event("SMB_GET", file, smb_info=attacker_IP)
FUNCTION monitor_smb_sessions(): // Thread 2 — Port 445 watcher
  LOOP every 2 s: Scan all TCP connections for port 445 ESTABLISHED
    IF new remote IP found: CALL log_event("SMB_ACCESS", "Network", ip)
    Update global active_smb_ips set
FUNCTION monitor_file_open(): // Thread 3 — Window title tracker
  LOOP every 0.5 s: Read foreground window title via win32gui
    IF title contains honeyfile keyword AND debounce(3 s) passes: CALL log_event("OPENED")
FUNCTION monitor_realtime_typing(): // Thread 4 — Keystroke interceptor
  ON each keypress: IF honeyfile window in focus AND debounce(2 s): CALL log_event("MODIFIED")
CLASS HoneyFileHandler (Watchdog FileSystemEventHandler): // Thread 5
  ON MODIFIED: Hash-compare file → CALL log_event("MODIFIED") or log_event("COPIED")
  ON DELETED: CALL log_event("DELETED") or log_event("CLOSED") // temp file = close event
  ON MOVED: Same directory → log_event("RENAMED") | Different dir → log_event("MOVED")
  ON CREATED: If temp/lock file → log_event("OPENED") | Else → add to copy candidates
FUNCTION process_copy_events(): // Thread 6 — MD5 hash copy detector
  LOOP every 2 s: FOR EACH copy candidate older than 2 s:
    IF MD5 hash matches any original HONEY_FILE hash: CALL log_event("COPIED")
// ── MAIN ENTRY POINT ────────────────────────────────────────────────────
MAIN:
  IF not Administrator → Re-launch via UAC ShellExecute (runas) → EXIT
  Prevent duplicate instances via Windows named Mutex
  Register pythonw.exe in Windows Startup registry key
  FOR EACH honeyfile in HONEY_FILES: Create file if missing (xlsx/pdf/pptx/docx/csv/txt)
  Compute MD5 hash of each honeyfile → store in ORIGINAL_HASHES
  START Watchdog observer on HONEY_DIR + spawn monitor_drives to watch all drive letters
  START threads: process_copy_events, monitor_file_open, monitor_realtime_typing,
                monitor_smb_sessions, monitor_security_logs // all daemon threads
  LOOP until KeyboardInterrupt → Stop observer → JOIN → EXIT
